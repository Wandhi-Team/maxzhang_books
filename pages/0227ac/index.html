获取目录下文件并重命名


$ext = "FHYQ";
$ext_rep = "-FHYQ";

//获取目录下所有文件
$files = scandir("./");



foreach ($files as $item) {
    if (strpos($item, "jpg")) {
        //文件重命名
        rename($item, str_replace($ext, $ext_rep, $item));
    }
}
