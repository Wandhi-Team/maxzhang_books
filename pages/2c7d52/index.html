css属性选择器的模糊匹配

# 模糊匹配

# ~=

包含匹配 （独立单词）

[title~=flower]  -->  <img src="/i/eg_tulip.jpg" title="tulip flower" />
1

# |= （独立单词）

以xx开头

[lang|=en]     -->  <p lang="en">  <p lang="en-us">
1

# ^=

以xx开头 （模糊包含）

[lang^=en]    -->  <p lang="ennn">
1

# $=

以xx结尾 （模糊包含）

a[src$=".pdf"]
1

# *=

包含匹配 （模糊包含）

[title*=flower]   -->  <img src="/i/eg_tulip.jpg" title="ffffflowerrrrrr" />
1
